function doPost(e) { const hoja = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("login"); const usuario = e.parameter.usuario; const clave = e.parameter.clave; const datos = hoja.getDataRange().getValues(); for (let i = 1; i < datos.length; i++) { const [usuarioHoja, claveHoja] = datos[i]; if (usuarioHoja === usuario && claveHoja === clave) { return ContentService.createTextOutput("ok"); } } return ContentService.createTextOutput("error"); }
